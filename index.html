        function updateCountdown() {

            const weddingDate = new Date('2025-06-15T17:00:00+08:00'); // 巴厘岛时间5:00 PM，6月15日

            const now = new Date();

            const timeDifference = weddingDate - now;


            if (timeDifference > 0) {

                const days = Math.floor(timeDifference / (1000 * 60 * 60 * 24));

                const hours = Math.floor((timeDifference % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));

                const minutes = Math.floor((timeDifference % (1000 * 60 * 60)) / (1000 * 60));

                

                const countdownElement = document.getElementById('countdown');

                if (countdownElement) {

                    if (days > 0) {

                        countdownElement.textContent = `${days}天 ${hours}小时 ${minutes}分钟`;

                    } else if (hours > 0) {

                        countdownElement.textContent = `${hours}小时 ${minutes}分钟`;

                    } else {

                        countdownElement.textContent = `${minutes}分钟`;
